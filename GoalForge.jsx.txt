// STORAGE HELPERS (window.storage with in-memory fallback)
// ══════════════════════════════════════════════════════════
const mem = {};
async function sGet(key) {
  try { const r = await window.storage.get(key); return r ? JSON.parse(r.value) : null; }
  catch { return mem[key] ?? null; }
}
async function sSet(key, val) {
  try { await window.storage.set(key, JSON.stringify(val)); mem[key] = val; }
  catch { mem[key] = val; }
}

// CONSTANTS

const GOAL_TYPES = [
  { value: "10d",      label: "10 Days",    days: 10  },
  { value: "15d",      label: "15 Days",    days: 15  },
  { value: "1m",       label: "1 Month",    days: 30  },
  { value: "3m",       label: "3 Months",   days: 90  },
  { value: "6m",       label: "6 Months",   days: 180 },
  { value: "1y",       label: "1 Year",     days: 365 },
  { value: "lifetime", label: "Lifetime ♾", days: null },
];

// ══════════════════════════════════════════════════════════
// PURE HELPERS
// ══════════════════════════════════════════════════════════
const uid = () => Math.random().toString(36).slice(2, 10);
const todayStr = () => new Date().toISOString().split("T")[0];
const daysBetween = (a, b) => Math.round((new Date(b) - new Date(a)) / 86400000);

function getDeadline(goal) {
  if (goal.deadline) return goal.deadline;
  const type = GOAL_TYPES.find(t => t.value === goal.type);
  if (!type?.days) return null;
  const d = new Date(goal.createdAt);
  d.setDate(d.getDate() + type.days);
  return d.toISOString().split("T")[0];
}

function getDaysRemaining(goal) {
  const dl = getDeadline(goal);
  if (!dl) return null;
  return daysBetween(todayStr(), dl);
}

function getOverallProgress(goal) {
  const cis = goal.checkIns || [];
  if (!cis.length) return 0;
  return Math.round(cis.reduce((s, c) => s + (c.percentage || 0), 0) / cis.length);
}

function getStreak(goal) {
  const cis = goal.checkIns || [];
  let streak = 0;
  const d = new Date();
  d.setDate(d.getDate() - 1);
  for (let i = 0; i < 1000; i++) {
    const expected = d.toISOString().split("T")[0];
    const ci = cis.find(c => c.date === expected);
    if (ci && !ci.missed) { streak++; d.setDate(d.getDate() - 1); }
    else break;
  }
  return streak;
}

function isOnTrack(goal) {
  const dl = getDeadline(goal);
  if (!dl) return true;
  const totalDays = daysBetween(goal.createdAt, dl);
  if (totalDays <= 0) return false;
  const elapsed = daysBetween(goal.createdAt, todayStr());
  const expected = (elapsed / totalDays) * 100;
  return getOverallProgress(goal) >= expected - 10;
}

async function askClaude(prompt) {
  const res = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      model: "claude-sonnet-4-20250514",
      max_tokens: 1000,
      messages: [{ role: "user", content: prompt }],
    }),
  });
  const data = await res.json();
  return data.content?.find(b => b.type === "text")?.text || "No response.";
}

// ══════════════════════════════════════════════════════════
// AUTH SCREEN
// ══════════════════════════════════════════════════════════
function AuthScreen({ t, dark, setDark, onLogin }) {
  const [mode, setMode]         = useState("login");
  const [username, setUsername] = useState("");
  const [password, setPassword] = useState("");
  const [error, setError]       = useState("");
  const [busy, setBusy]         = useState(false);

  const submit = async () => {
    if (!username.trim() || !password.trim()) { setError("Fill all fields."); return; }
    setBusy(true); setError("");
    const users = await sGet("users") || {};
    if (mode === "register") {
      if (users[username]) { setError("Username taken."); setBusy(false); return; }
      users[username] = { password, createdAt: new Date().toISOString() };
      await sSet("users", users);
      onLogin({ username });
    } else {
      if (!users[username] || users[username].password !== password) {
        setError("Wrong username or password."); setBusy(false); return;
      }
      onLogin({ username });
    }
    setBusy(false);
  };

  return (
    <div style={{ display: "flex", alignItems: "center", justifyContent: "center", minHeight: "100vh", padding: 20, background: t.bg }}>
      <div style={{ width: "100%", maxWidth: 400 }}>
        {/* Logo */}
        <div style={{ textAlign: "center", marginBottom: 36 }}>
          <div style={{ fontSize: 52, marginBottom: 8 }}>⚡</div>
          <h1 style={{ fontFamily: "'Syne',sans-serif", fontWeight: 800, fontSize: 34, color: t.accent, letterSpacing: "-1px" }}>GoalForge</h1>
          <p style={{ color: t.sub, fontSize: 14, marginTop: 5 }}>Set. Track. Achieve. Repeat.</p>
          <div style={{ marginTop: 14 }}>
            <button onClick={() => setDark(!dark)} style={{ background: t.card, border: `1px solid ${t.border}`, borderRadius: 20, padding: "6px 14px", cursor: "pointer", fontSize: 14 }}>
              {dark ? "☀️ Light" : "🌙 Dark"}
            </button>
          </div>
        </div>

        <div style={{ background: t.surface, borderRadius: 22, padding: "28px 24px", border: `1px solid ${t.border}` }}>
          {/* Tabs */}
          <div style={{ display: "flex", background: t.card, borderRadius: 12, padding: 4, marginBottom: 22, gap: 4 }}>
            {["login", "register"].map(m => (
              <button key={m} onClick={() => setMode(m)} style={{
                flex: 1, padding: 10, borderRadius: 10, border: "none", cursor: "pointer",
                background: mode === m ? t.accent : "transparent",
                color: mode === m ? "#fff" : t.sub,
                fontWeight: 600, fontSize: 14, fontFamily: "'Outfit',sans-serif", transition: "all 0.2s",
              }}>{m === "login" ? "Login" : "Register"}</button>
            ))}
          </div>

          {/* Fields */}
          <div style={{ display: "flex", flexDirection: "column", gap: 12 }}>
            <Field t={t} icon="👤" placeholder="Username" value={username} onChange={e => setUsername(e.target.value)} />
            <Field t={t} icon="🔒" placeholder="Password" value={password} onChange={e => setPassword(e.target.value)} type="password"
              onKeyDown={e => e.key === "Enter" && submit()} />
            {error && <p style={{ color: t.danger, fontSize: 13, textAlign: "center" }}>{error}</p>}
            <Btn t={t} onClick={submit} disabled={busy} style={{ marginTop: 4 }}>
              {busy ? "..." : mode === "login" ? "Login →" : "Create Account →"}
            </Btn>
          </div>
        </div>
      </div>
    </div>
  );
}

// DASHBOARD
// ══════════════════════════════════════════════════════════
function Dashboard({ t, goals, onAdd, onSelect }) {
  const today = todayStr();
  const pending = goals.filter(g => !(g.checkIns || []).some(c => c.date === today));
  const totalStreak = goals.reduce((s, g) => s + getStreak(g), 0);
  const avgProg = goals.length ? Math.round(goals.reduce((s, g) => s + getOverallProgress(g), 0) / goals.length) : 0;

  return (
    <div style={{ maxWidth: 860, margin: "0 auto", padding: "22px 18px" }}>
      {/* Stats */}
      <div style={{ display: "grid", gridTemplateColumns: "repeat(3,1fr)", gap: 12, marginBottom: 22 }}>
        <StatCard t={t} label="Goals" value={goals.length} icon="🎯" color={t.accent} />
        <StatCard t={t} label="Streak Days" value={`${totalStreak}`} icon="🔥" color={t.warn} />
        <StatCard t={t} label="Avg Progress" value={`${avgProg}%`} icon="📈" color={t.green} />
      </div>

      {/* Reminder */}
      {pending.length > 0 && (
        <div style={{
          background: `linear-gradient(135deg, rgba(255,170,0,0.13), rgba(255,100,0,0.08))`,
          border: `1px solid ${t.warn}50`, borderRadius: 14, padding: "13px 18px",
          marginBottom: 18, display: "flex", alignItems: "center", gap: 10,
        }}>
          <span style={{ fontSize: 22 }}>⏰</span>
          <div>
            <p style={{ fontWeight: 700, color: t.warn, fontSize: 14 }}>Check-in Reminder</p>
            <p style={{ color: t.sub, fontSize: 13 }}>{pending.length} goal{pending.length > 1 ? "s" : ""} need today's check-in</p>
          </div>
        </div>
      )}

      {/* Header */}
      <div style={{ display: "flex", alignItems: "center", justifyContent: "space-between", marginBottom: 14 }}>
        <h2 style={{ fontFamily: "'Syne',sans-serif", fontWeight: 800, fontSize: 20 }}>My Goals</h2>
        <Btn t={t} onClick={onAdd} small>+ New Goal</Btn>
      </div>

      {/* List */}
      {goals.length === 0 ? (
        <div style={{ textAlign: "center", padding: "56px 20px", color: t.sub }}>
          <div style={{ fontSize: 52, marginBottom: 12 }}>🎯</div>
          <p style={{ fontSize: 18, fontWeight: 700, marginBottom: 8, color: t.text }}>No goals yet</p>
          <p style={{ fontSize: 14 }}>Tap "+ New Goal" to start forging your first goal.</p>
        </div>
      ) : (
        <div style={{ display: "flex", flexDirection: "column", gap: 12 }}>
          {goals.map(g => <GoalCard key={g.id} t={t} goal={g} today={today} onClick={() => onSelect(g.id)} />)}
        </div>
      )}
    </div>
  );
}

function StatCard({ t, label, value, icon, color }) {
  return (
    <div style={{ background: t.surface, border: `1px solid ${t.border}`, borderRadius: 16, padding: "16px 12px", textAlign: "center" }}>
      <div style={{ fontSize: 22, marginBottom: 5 }}>{icon}</div>
      <div style={{ fontWeight: 800, fontSize: 22, color, fontFamily: "'Syne',sans-serif" }}>{value}</div>
      <div style={{ color: t.sub, fontSize: 11, marginTop: 2 }}>{label}</div>
    </div>
  );
}

function GoalCard({ t, goal, today, onClick }) {
  const daysLeft = getDaysRemaining(goal);
  const progress = getOverallProgress(goal);
  const streak   = getStreak(goal);
  const onTrack  = isOnTrack(goal);
  const hasTodayCI   = (goal.checkIns || []).some(c => c.date === today);
  const isNear   = daysLeft !== null && daysLeft <= 3 && daysLeft >= 0;
  const isOver   = daysLeft !== null && daysLeft < 0;

  return (
    <div onClick={onClick} className="hover-lift press" style={{
      background: t.surface, border: `1px solid ${isOver ? t.danger + "50" : isNear ? t.warn + "50" : t.border}`,
      borderRadius: 16, padding: "18px 18px", cursor: "pointer",
    }}>
      {/* Badges */}
      <div style={{ display: "flex", gap: 6, flexWrap: "wrap", marginBottom: 9 }}>
        <Badge color={t.accent} t={t}>{GOAL_TYPES.find(gt => gt.value === goal.type)?.label}</Badge>
        <Badge color={onTrack ? t.green : t.danger} t={t}>{onTrack ? "✓ On Track" : "⚠ Behind"}</Badge>
        {!hasTodayCI && <Badge color={t.warn} t={t}>📝 Check-in due</Badge>}
        {isNear && !isOver && <Badge color={t.warn} t={t}>⏰ Deadline near!</Badge>}
        {isOver && <Badge color={t.danger} t={t}>❌ Overdue</Badge>}
      </div>

      <div style={{ display: "flex", alignItems: "flex-start", justifyContent: "space-between", gap: 12 }}>
        <div style={{ flex: 1 }}>
          <h3 style={{ fontWeight: 700, fontSize: 15, marginBottom: 12, lineHeight: 1.35 }}>{goal.title}</h3>
          {/* Progress bar */}
          <div>
            <div style={{ display: "flex", justifyContent: "space-between", fontSize: 12, color: t.sub, marginBottom: 5 }}>
              <span>Progress</span>
              <span style={{ fontWeight: 700, color: t.text }}>{progress}%</span>
            </div>
            <div style={{ height: 5, background: t.border, borderRadius: 3 }}>
              <div style={{ height: "100%", width: `${progress}%`, background: `linear-gradient(90deg,${t.accent},${t.green})`, borderRadius: 3, transition: "width 0.5s" }} />
            </div>
          </div>
        </div>

        {/* Right side */}
        <div style={{ display: "flex", flexDirection: "column", alignItems: "center", gap: 6, minWidth: 60 }}>
          <div style={{ textAlign: "center", background: t.card, borderRadius: 10, padding: "8px 10px" }}>
            <div style={{ fontSize: 16 }}>🔥</div>
            <div style={{ fontWeight: 700, fontSize: 14, color: t.warn }}>{streak}</div>
            <div style={{ fontSize: 10, color: t.sub }}>streak</div>
          </div>
          {daysLeft !== null && (
            <div style={{ fontSize: 12, fontWeight: 700, color: isOver ? t.danger : isNear ? t.warn : t.sub, textAlign: "center" }}>
              {isOver ? "Overdue" : `${daysLeft}d left`}
            </div>
          )}
        </div>
      </div>
    </div>
  );
}

// ══════════════════════════════════════════════════════════
// ADD GOAL SCREEN
// ══════════════════════════════════════════════════════════
function AddGoalScreen({ t, user, onSave }) {
  const [title, setTitle]     = useState("");
  const [type, setType]       = useState("1m");
  const [deadline, setDeadline] = useState("");
  const [subTasks, setSubTasks] = useState([]);
  const [newTask, setNewTask]  = useState("");
  const [error, setError]     = useState("");

  const addTask = () => {
    if (!newTask.trim()) return;
    setSubTasks([...subTasks, { id: uid(), text: newTask.trim() }]);
    setNewTask("");
  };

  const save = () => {
    if (!title.trim()) { setError("Enter a goal title."); return; }
    const typeObj = GOAL_TYPES.find(gt => gt.value === type);
    const today = todayStr();
    let dl = deadline || null;
    if (!dl && typeObj?.days) {
      const d = new Date();
      d.setDate(d.getDate() + typeObj.days);
      dl = d.toISOString().split("T")[0];
    }
    onSave({ id: uid(), userId: user.username, title: title.trim(), type, deadline: dl, createdAt: today, subTasks, checkIns: [], userWeaknesses: "" });
  };

  return (
    <div style={{ maxWidth: 600, margin: "0 auto", padding: "24px 18px" }}>
      <h2 style={{ fontFamily: "'Syne',sans-serif", fontWeight: 800, fontSize: 22, marginBottom: 24 }}>🎯 New Goal</h2>

      <div style={{ display: "flex", flexDirection: "column", gap: 20 }}>
        {/* Title */}
        <Section t={t} label="Goal Title">
          <Field t={t} icon="📌" placeholder="e.g. Clear my backlog exam" value={title} onChange={e => setTitle(e.target.value)} />
        </Section>

        {/* Type */}
        <Section t={t} label="Goal Type">
          <div style={{ display: "grid", gridTemplateColumns: "repeat(auto-fill,minmax(105px,1fr))", gap: 8 }}>
            {GOAL_TYPES.map(gt => (
              <button key={gt.value} onClick={() => setType(gt.value)} style={{
                padding: "10px 6px", borderRadius: 10,
                border: `1px solid ${type === gt.value ? t.accent : t.border}`,
                background: type === gt.value ? `${t.accent}18` : t.surface,
                color: type === gt.value ? t.accent : t.sub,
                fontWeight: 600, fontSize: 13, cursor: "pointer", fontFamily: "'Outfit',sans-serif", transition: "all 0.18s",
              }}>{gt.label}</button>
            ))}
          </div>
        </Section>

        {/* Deadline */}
        <Section t={t} label="Custom Deadline (optional)">
          <Field t={t} icon="📅" type="date" value={deadline} min={todayStr()}
            onChange={e => setDeadline(e.target.value)} placeholder="Select date" />
        </Section>

        {/* Sub-tasks */}
        <Section t={t} label="Daily Checklist Tasks (optional)">
          <div style={{ display: "flex", gap: 8, marginBottom: 8 }}>
            <Field t={t} icon="✏️" placeholder="Add a daily task..." value={newTask}
              onChange={e => setNewTask(e.target.value)}
              onKeyDown={e => e.key === "Enter" && addTask()} style={{ flex: 1 }} />
            <button onClick={addTask} style={{
              background: t.accent, color: "#fff", border: "none", borderRadius: 12,
              padding: "0 16px", cursor: "pointer", fontWeight: 700, fontSize: 20, flexShrink: 0,
            }}>+</button>
          </div>
          {subTasks.length > 0 && (
            <div style={{ display: "flex", flexDirection: "column", gap: 6 }}>
              {subTasks.map((st, i) => (
                <div key={st.id} style={{ display: "flex", alignItems: "center", gap: 10, background: t.card, borderRadius: 10, padding: "10px 14px", border: `1px solid ${t.border}` }}>
                  <span style={{ color: t.accent, fontSize: 11, fontWeight: 700, minWidth: 20 }}>#{i + 1}</span>
                  <span style={{ flex: 1, fontSize: 14 }}>{st.text}</span>
                </div>
              ))}
            </div>
          )}
          <p style={{ color: t.sub, fontSize: 12, marginTop: 6 }}>These tasks appear in your nightly check-in.</p>
        </Section>

        {error && <p style={{ color: t.danger, fontSize: 13 }}>{error}</p>}

        <Btn t={t} onClick={save} style={{ marginTop: 4 }}>Save Goal ⚡</Btn>
      </div>
    </div>
  );
}

// ══════════════════════════════════════════════════════════
// GOAL DETAIL SCREEN
// ══════════════════════════════════════════════════════════
function GoalDetailScreen({ t, goal, onUpdate }) {
  const today         = todayStr();
  const daysLeft      = getDaysRemaining(goal);
  const progress      = getOverallProgress(goal);
  const streak        = getStreak(goal);
  const onTrack       = isOnTrack(goal);
  const hasTodayCI    = (goal.checkIns || []).some(c => c.date === today);
  const isNear        = daysLeft !== null && daysLeft <= 3 && daysLeft >= 0;
  const isOver        = daysLeft !== null && daysLeft < 0;

  const [showCI, setShowCI]           = useState(false);
  const [aiSug, setAiSug]             = useState("");
  const [sугLoad, setSugLoad]         = useState(false);
  const [aiWeak, setAiWeak]           = useState("");
  const [weakLoad, setWeakLoad]       = useState(false);
  const [userWeak, setUserWeak]       = useState(goal.userWeaknesses || "");
  const [weakSaved, setWeakSaved]     = useState(false);

  // Chart: last 14 check-ins
  const chartData = [...(goal.checkIns || [])]
    .sort((a, b) => a.date.localeCompare(b.date))
    .slice(-14)
    .map(ci => ({ date: ci.date.slice(5), pct: ci.missed ? 0 : (ci.percentage || 0), missed: ci.missed }));

  const getSuggestion = async () => {
    setSugLoad(true);
    const notes = (goal.checkIns || []).slice(-5).map(c => c.note || "").filter(Boolean).join(". ");
    const prompt = `Goal: "${goal.title}" (${goal.type}, ${daysLeft ?? "lifetime"} days left). Progress: ${progress}%. Status: ${onTrack ? "on track" : "behind"}. Recent notes: "${notes}". Give 2-3 specific, actionable things to do TODAY to achieve this goal. Be direct and brief.`;
    setAiSug(await askClaude(prompt));
    setSugLoad(false);
  };

  const analyzeWeakness = async () => {
    setWeakLoad(true);
    const notes = (goal.checkIns || []).map(c => `${c.date}: ${c.note || "no note"} (${c.percentage}%)`).join("\n");
    const prompt = `Goal: "${goal.title}". My self-noted weaknesses: "${userWeak}". Check-in log:\n${notes || "No check-ins yet"}\nIdentify 2-3 key patterns or weaknesses holding me back. Be specific and constructive.`;
    setAiWeak(await askClaude(prompt));
    setWeakLoad(false);
  };

  const saveWeak = () => {
    onUpdate({ ...goal, userWeaknesses: userWeak });
    setWeakSaved(true);
    setTimeout(() => setWeakSaved(false), 2000);
  };

  return (
    <div style={{ maxWidth: 820, margin: "0 auto", padding: "22px 18px" }}>
      {/* Header */}
      <div style={{ marginBottom: 20 }}>
        <div style={{ display: "flex", gap: 6, flexWrap: "wrap", marginBottom: 8 }}>
          <Badge color={t.accent} t={t}>{GOAL_TYPES.find(gt => gt.value === goal.type)?.label}</Badge>
          <Badge color={onTrack ? t.green : t.danger} t={t}>{onTrack ? "✓ On Track" : "⚠ Behind Schedule"}</Badge>
          {isNear && !isOver && <Badge color={t.warn} t={t}>⏰ Deadline Near!</Badge>}
          {isOver && <Badge color={t.danger} t={t}>❌ Overdue</Badge>}
        </div>
        <h1 style={{ fontFamily: "'Syne',sans-serif", fontWeight: 800, fontSize: 24, lineHeight: 1.25 }}>{goal.title}</h1>
        <p style={{ color: t.sub, fontSize: 13, marginTop: 5 }}>Started: {goal.createdAt}{goal.deadline ? ` · Deadline: ${goal.deadline}` : ""}</p>
      </div>

      {/* Stats row */}
      <div style={{ display: "grid", gridTemplateColumns: "repeat(3,1fr)", gap: 12, marginBottom: 20 }}>
        <StatMini t={t} label="Progress" value={`${progress}%`} color={t.accent} bar={progress} />
        <StatMini t={t} label="Streak" value={`🔥 ${streak}`} color={t.warn} />
        <StatMini t={t} label="Days Left" value={daysLeft === null ? "♾" : isOver ? "Overdue" : `${daysLeft}`}
          color={isOver ? t.danger : isNear ? t.warn : t.green} />
      </div>

      {/* Check-in button */}
      {!hasTodayCI ? (
        <button onClick={() => setShowCI(true)} className="press" style={{
          width: "100%", background: `linear-gradient(135deg, ${t.orange}, ${t.danger})`,
          color: "#fff", border: "none", borderRadius: 14, padding: "15px",
          fontSize: 15, fontWeight: 700, cursor: "pointer", marginBottom: 20, fontFamily: "'Outfit',sans-serif",
        }}>📝 Tonight's Check-in (9:30 PM)</button>
      ) : (
        <div style={{ background: `${t.green}12`, border: `1px solid ${t.green}40`, borderRadius: 12, padding: "12px 18px", marginBottom: 20, textAlign: "center" }}>
          <span style={{ color: t.green, fontWeight: 600, fontSize: 14 }}>✓ Today's check-in is done!</span>
        </div>
      )}

      {/* AI Suggestion */}
      <div style={{ background: t.surface, border: `1px solid ${t.border}`, borderRadius: 16, padding: 18, marginBottom: 16 }}>
        <div style={{ display: "flex", alignItems: "center", justifyContent: "space-between", marginBottom: 10 }}>
          <h3 style={{ fontWeight: 700, fontSize: 14 }}>🤖 AI Suggestion</h3>
          <SmallBtn t={t} color={t.accent} onClick={getSuggestion} loading={sугLoad}>
            {sугLoad ? "Thinking..." : "Get Advice"}
          </SmallBtn>
        </div>
        {aiSug
          ? <p style={{ color: t.text, fontSize: 14, lineHeight: 1.75, whiteSpace: "pre-wrap" }}>{aiSug}</p>
          : <p style={{ color: t.sub, fontSize: 13 }}>Click "Get Advice" for AI-powered daily action plan.</p>
        }
      </div>

      {/* Bar Chart */}
      {chartData.length > 0 && (
        <div style={{ background: t.surface, border: `1px solid ${t.border}`, borderRadius: 16, padding: 18, marginBottom: 16 }}>
          <h3 style={{ fontWeight: 700, fontSize: 14, marginBottom: 14 }}>📊 Daily Progress (last {chartData.length} days)</h3>
          <ResponsiveContainer width="100%" height={150}>
            <BarChart data={chartData} barSize={16} margin={{ top: 0, right: 0, bottom: 0, left: -10 }}>
              <XAxis dataKey="date" tick={{ fontSize: 10, fill: t.sub }} axisLine={false} tickLine={false} />
              <YAxis domain={[0, 100]} tick={{ fontSize: 10, fill: t.sub }} axisLine={false} tickLine={false} />
              <Tooltip
                contentStyle={{ background: t.card, border: `1px solid ${t.border}`, borderRadius: 8, fontSize: 12, color: t.text }}
                formatter={v => [`${v}%`, "Progress"]}
                labelStyle={{ color: t.sub }}
              />
              <Bar dataKey="pct" radius={[4, 4, 0, 0]}>
                {chartData.map((e, i) => (
                  <Cell key={i} fill={e.missed ? t.danger : e.pct >= 80 ? t.green : e.pct >= 50 ? t.accent : t.warn} />
                ))}
              </Bar>
            </BarChart>
          </ResponsiveContainer>
          <div style={{ display: "flex", gap: 14, marginTop: 8, flexWrap: "wrap" }}>
            {[{ c: t.green, l: "≥80%" }, { c: t.accent, l: "50–79%" }, { c: t.warn, l: "<50%" }, { c: t.danger, l: "Missed" }].map(({ c, l }) => (
              <div key={l} style={{ display: "flex", alignItems: "center", gap: 5, fontSize: 11, color: t.sub }}>
                <div style={{ width: 9, height: 9, borderRadius: 2, background: c }} />{l}
              </div>
            ))}
          </div>
        </div>
      )}

      {/* Check-in History */}
      {(goal.checkIns || []).length > 0 && (
        <div style={{ background: t.surface, border: `1px solid ${t.border}`, borderRadius: 16, padding: 18, marginBottom: 16 }}>
          <h3 style={{ fontWeight: 700, fontSize: 14, marginBottom: 12 }}>📋 Check-in History</h3>
          <div style={{ display: "flex", flexDirection: "column", gap: 8, maxHeight: 280, overflowY: "auto" }}>
            {[...(goal.checkIns || [])].sort((a, b) => b.date.localeCompare(a.date)).map((ci, i) => (
              <CIHistoryItem key={i} ci={ci} t={t} subTasks={goal.subTasks || []} />
            ))}
          </div>
        </div>
      )}

      {/* Weaknesses */}
      <div style={{ background: t.surface, border: `1px solid ${t.border}`, borderRadius: 16, padding: 18, marginBottom: 20 }}>
        <h3 style={{ fontWeight: 700, fontSize: 14, marginBottom: 12 }}>🔍 Weaknesses & Cons</h3>
        <textarea
          value={userWeak} onChange={e => setUserWeak(e.target.value)}
          placeholder="Write your own weaknesses... e.g. I procrastinate at night, I get distracted by phone..."
          style={{
            width: "100%", background: t.card, border: `1px solid ${t.border}`,
            borderRadius: 10, padding: "12px 14px", color: t.text, fontSize: 14,
            fontFamily: "'Outfit',sans-serif", resize: "vertical", minHeight: 80, lineHeight: 1.6,
          }}
        />
        <div style={{ display: "flex", gap: 8, marginTop: 10, flexWrap: "wrap" }}>
          <SmallBtn t={t} color={t.green} onClick={saveWeak}>{weakSaved ? "✓ Saved!" : "Save"}</SmallBtn>
          <SmallBtn t={t} color={t.purple} onClick={analyzeWeakness} loading={weakLoad}>
            {weakLoad ? "Analyzing..." : "🤖 AI Analyze"}
          </SmallBtn>
        </div>
        {aiWeak && (
          <div style={{ marginTop: 14, background: t.card, borderRadius: 10, padding: "14px 16px", border: `1px solid ${t.border}` }}>
            <p style={{ color: t.sub, fontSize: 11, fontWeight: 700, marginBottom: 6, textTransform: "uppercase", letterSpacing: "0.5px" }}>AI Analysis</p>
            <p style={{ color: t.text, fontSize: 14, lineHeight: 1.75, whiteSpace: "pre-wrap" }}>{aiWeak}</p>
          </div>
        )}
      </div>

      {/* Check-in Modal */}
      {showCI && (
        <CheckInModal t={t} goal={goal}
          onClose={() => setShowCI(false)}
          onSubmit={ci => { onUpdate({ ...goal, checkIns: [...(goal.checkIns || []), ci] }); setShowCI(false); }}
        />
      )}
    </div>
  );
}

function StatMini({ t, label, value, color, bar }) {
  return (
    <div style={{ background: t.surface, border: `1px solid ${t.border}`, borderRadius: 14, padding: "14px 12px", textAlign: "center" }}>
      <div style={{ fontWeight: 800, fontSize: 22, color, fontFamily: "'Syne',sans-serif" }}>{value}</div>
      <div style={{ color: t.sub, fontSize: 11, marginTop: 3 }}>{label}</div>
      {bar !== undefined && (
        <div style={{ height: 3, background: t.border, borderRadius: 2, marginTop: 8 }}>
          <div style={{ height: "100%", width: `${bar}%`, background: color, borderRadius: 2, transition: "width 0.5s" }} />
        </div>
      )}
    </div>
  );
}

function CIHistoryItem({ ci, t, subTasks }) {
  const completed = ci.completedTasks?.length || 0;
  const total = subTasks.length;
  return (
    <div style={{ background: t.card, borderRadius: 10, padding: "11px 14px", border: `1px solid ${ci.missed ? t.danger + "30" : t.border}` }}>
      <div style={{ display: "flex", alignItems: "center", justifyContent: "space-between" }}>
        <div style={{ display: "flex", alignItems: "center", gap: 8 }}>
          <span style={{ fontSize: 12, color: t.sub }}>{ci.date}</span>
          {ci.missed && <span style={{ color: t.danger, fontSize: 10, fontWeight: 700, background: t.danger + "20", padding: "2px 7px", borderRadius: 6 }}>MISSED</span>}
          {total > 0 && !ci.missed && <span style={{ fontSize: 11, color: t.sub }}>{completed}/{total} tasks</span>}
        </div>
        <span style={{ fontWeight: 700, fontSize: 14, color: ci.missed ? t.danger : ci.percentage >= 80 ? t.green : ci.percentage >= 50 ? t.accent : t.warn }}>
          {ci.missed ? "0%" : `${ci.percentage}%`}
        </span>
      </div>
      {ci.note && <p style={{ color: t.sub, fontSize: 13, marginTop: 5, lineHeight: 1.5 }}>{ci.note}</p>}
    </div>
  );
}

// ══════════════════════════════════════════════════════════
// CHECK-IN MODAL
// ══════════════════════════════════════════════════════════
function CheckInModal({ t, goal, onClose, onSubmit }) {
  const subTasks = goal.subTasks || [];
  const [checked, setChecked] = useState({});
  const [note, setNote]       = useState("");

  const done = Object.values(checked).filter(Boolean).length;
  const pct  = subTasks.length > 0 ? Math.round((done / subTasks.length) * 100) : 100;

  const toggle = id => setChecked(c => ({ ...c, [id]: !c[id] }));

  const submit = () => {
    onSubmit({
      date: todayStr(),
      completedTasks: subTasks.filter(st => checked[st.id]).map(st => st.id),
      note: note.trim(),
      percentage: pct,
      missed: false,
    });
  };

  return (
    <div style={{
      position: "fixed", inset: 0, background: "rgba(0,0,0,0.82)",
      display: "flex", alignItems: "center", justifyContent: "center",
      zIndex: 500, padding: 18,
    }} onClick={e => e.target === e.currentTarget && onClose()}>
      <div style={{
        background: t.surface, borderRadius: 22, padding: "26px 22px",
        width: "100%", maxWidth: 480, border: `1px solid ${t.border}`,
        animation: "pop 0.25s ease", maxHeight: "90vh", overflowY: "auto",
      }}>
        <div style={{ display: "flex", alignItems: "center", justifyContent: "space-between", marginBottom: 18 }}>
          <h2 style={{ fontFamily: "'Syne',sans-serif", fontWeight: 800, fontSize: 18 }}>📝 Check-in</h2>
          <button onClick={onClose} style={{ background: "none", border: "none", color: t.sub, cursor: "pointer", fontSize: 20 }}>✕</button>
        </div>

        {/* Big % */}
        <div style={{ textAlign: "center", padding: "16px", background: t.card, borderRadius: 14, marginBottom: 18 }}>
          <div style={{ fontWeight: 900, fontSize: 48, color: pct >= 80 ? t.green : pct >= 50 ? t.accent : t.warn, fontFamily: "'Syne',sans-serif", lineHeight: 1 }}>{pct}%</div>
          <div style={{ color: t.sub, fontSize: 13, marginTop: 4 }}>Today's Progress</div>
          <div style={{ height: 5, background: t.border, borderRadius: 3, marginTop: 10 }}>
            <div style={{ height: "100%", width: `${pct}%`, background: `linear-gradient(90deg,${t.accent},${t.green})`, borderRadius: 3, transition: "width 0.3s" }} />
          </div>
        </div>

        {/* Checklist */}
        {subTasks.length > 0 && (
          <div style={{ marginBottom: 16 }}>
            <p style={{ fontWeight: 700, fontSize: 11, color: t.sub, marginBottom: 10, textTransform: "uppercase", letterSpacing: "0.6px" }}>Daily Tasks</p>
            <div style={{ display: "flex", flexDirection: "column", gap: 7 }}>
              {subTasks.map(st => (
                <div key={st.id} onClick={() => toggle(st.id)} style={{
                  display: "flex", alignItems: "center", gap: 12, background: t.card,
                  borderRadius: 10, padding: "11px 13px", cursor: "pointer",
                  border: `1px solid ${checked[st.id] ? t.green + "60" : t.border}`,
                  transition: "border-color 0.18s",
                }}>
                  <div style={{
                    width: 20, height: 20, borderRadius: 6, flexShrink: 0,
                    border: `2px solid ${checked[st.id] ? t.green : t.border}`,
                    background: checked[st.id] ? t.green : "transparent",
                    display: "flex", alignItems: "center", justifyContent: "center",
                    transition: "all 0.18s",
                  }}>
                    {checked[st.id] && <span style={{ color: "#fff", fontSize: 11, fontWeight: 700 }}>✓</span>}
                  </div>
                  <span style={{ fontSize: 14, color: checked[st.id] ? t.sub : t.text, textDecoration: checked[st.id] ? "line-through" : "none", transition: "color 0.15s" }}>{st.text}</span>
                </div>
              ))}
            </div>
          </div>
        )}

        {subTasks.length === 0 && (
          <div style={{ textAlign: "center", padding: "14px", background: t.card, borderRadius: 10, marginBottom: 16 }}>
            <p style={{ color: t.sub, fontSize: 13 }}>No tasks for this goal — progress set to 100%.</p>
          </div>
        )}

        {/* Note */}
        <div style={{ marginBottom: 18 }}>
          <p style={{ fontWeight: 700, fontSize: 11, color: t.sub, marginBottom: 8, textTransform: "uppercase", letterSpacing: "0.6px" }}>What did you do today?</p>
          <textarea
            value={note} onChange={e => setNote(e.target.value)}
            placeholder="Write what you did for this goal today..."
            style={{
              width: "100%", background: t.card, border: `1px solid ${t.border}`,
              borderRadius: 10, padding: "12px 14px", color: t.text, fontSize: 14,
              fontFamily: "'Outfit',sans-serif", resize: "vertical", minHeight: 80, lineHeight: 1.6,
            }}
          />
        </div>

        <Btn t={t} onClick={submit}>Submit Check-in ✓</Btn>
      </div>
    </div>
  );
}

function Field({ t, icon, placeholder, value, onChange, type = "text", onKeyDown, min, style }) {
  return (
    <div style={{ display: "flex", alignItems: "center", gap: 10, background: t.surface, borderRadius: 12, padding: "11px 14px", border: `1px solid ${t.border}`, ...style }}>
      {icon && <span style={{ fontSize: 15, flexShrink: 0 }}>{icon}</span>}
      <input
        type={type} placeholder={placeholder} value={value} onChange={onChange}
        onKeyDown={onKeyDown} min={min}
        style={{ flex: 1, background: "none", border: "none", color: t.text, fontSize: 15, fontFamily: "'Outfit',sans-serif" }}
      />
    </div>
  );
}

function Btn({ t, onClick, children, disabled, small, style }) {
  return (
    <button onClick={onClick} disabled={disabled} className="press" style={{
      background: `linear-gradient(135deg, ${t.accent}, #005fcc)`,
      color: "#fff", border: "none", borderRadius: small ? 10 : 13,
      padding: small ? "8px 16px" : "14px 20px",
      fontSize: small ? 13 : 15, fontWeight: 700, cursor: "pointer",
      fontFamily: "'Outfit',sans-serif", opacity: disabled ? 0.65 : 1,
      width: small ? "auto" : "100%", ...style,
    }}>{children}</button>
  );
}

function SmallBtn({ t, color, onClick, loading, children }) {
  return (
    <button onClick={onClick} disabled={loading} className="press" style={{
      background: `${color}18`, color, border: `1px solid ${color}40`,
      borderRadius: 8, padding: "7px 14px", fontSize: 13, fontWeight: 600,
      cursor: "pointer", fontFamily: "'Outfit',sans-serif", opacity: loading ? 0.6 : 1,
    }}>{children}</button>
  );
}

function Badge({ color, t, children }) {
  return (
    <span style={{
      background: `${color}18`, color, fontSize: 11, fontWeight: 600,
      padding: "3px 10px", borderRadius: 20, border: `1px solid ${color}30`,
    }}>{children}</span>
  );
}

function Section({ t, label, children }) {
  return (
    <div>
      <p style={{ fontSize: 12, fontWeight: 700, color: t.sub, marginBottom: 8, textTransform: "uppercase", letterSpacing: "0.5px" }}>{label}</p>
      {children}
    </div>
  );
}
